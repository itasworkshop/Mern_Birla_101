SQL --> structured query language
to interact with data as well as schema

create table emp(emp_id number, ename varchar2(30), salary number, dept_id number);

describe emp;

insert into emp values(101,'Rajesh',50000,111);
insert into emp values(103,'Tom',60000,222);

select emp_id,ename,salary,dept_id from emp;
select emp_id,ename from emp;
select * from emp;

select * from emp where dept_id = 111;
select * from emp where dept_id = 111 and ename like 'Suraj';

select * from emp order by emp_id asc;
select ename,salary ,(salary+500) as bonus from emp;

DML--> Data Manipulation language
update emp set salary=70000 where ename like 'Rajesh'; 
delete from emp where emp_id = 102;
