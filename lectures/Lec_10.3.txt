5) For fetching you can use useEffect Hook

useEffect
ans: to perform side effects on our components, for ex - fetching data, dom update etc.

function Timer(){
  const[count,setCount] = useState(0);

  useEffect(() => {
    setTimeout(() => {
      setCount(count => count +1);
    }, 1000);
  });

  return <h1>The current counter is updated {count} times.</h1>
}

render(
  <Timer />, document.querySelector('#root')
)
