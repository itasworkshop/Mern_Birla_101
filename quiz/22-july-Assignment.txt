
//1)WAP to check whether given matrix is identical matrix or not for 2*2 and 3*3 matrix?
function checkMatrix(data) {
  for (let i = 0; i < data.length; i++) 
   {
    for (let j = 0; j < data.length; j++)
     {
      if (data[i][j] !== 1 && i === j )
      {
        return "it is not an identical";
      }
    }
  }
  return "it is an identical";
}

console.log("3*3 Matrix is: " + checkMatrix([[1, 0, 0],  [0, 1, 0],  [0, 0, 1]]))
console.log("2*2 Matrix is: " + checkMatrix([[1, 0],  [0, 1]]))
/*
OutPut: 
3*3 Matrix is: true
2*2 Matrix is: true 
Explanation:

forLoop for iterating the nestedarray, I've created the logic to check if the array of row and column is equal, And also check if diagonal values of array is not equal to 1, It will return "it is not an identical", Else It will return "it is an identical" , 
*/

//4) WAP to check whether given number or string is palindrom or not?
var string = "madam";
function checkPalindromeOrNot(str) {
    for (let i = 0; i < str.length / 2; i++) {
        if (str[i] !== str[str.length - 1 - i]) {
            return 'It is not a palindrome';
        }
    }
    return 'It is a palindrome';
}

console.log(`String Is : ${string}` );
console.log(checkPalindromeOrNot(string));
